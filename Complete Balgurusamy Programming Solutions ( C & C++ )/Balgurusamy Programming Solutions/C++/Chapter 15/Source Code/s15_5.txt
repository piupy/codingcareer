#include<iostream>
#include<cstdio>
#include<cstring>
#include<conio.h>
using namespace std;

int main(void)
{
	string s;
	cout<<"Enter The Line of Text :    ";
	int i,count=0;
	char find;
	getline(cin,s);
	cout<<"\n\nYou Entered :   "<<s;
	cout<<"\n\nEnter The Character You Want To Search :    ";
	cin>>find;
		for(i=0;i<s.length();i++)
		{
			if( s.at(i) == find)
				count++;
		}
	if( count == 0 )
		cout<<"\n\n\'"<<find<<"\' Not Found in The Line...\n\n";
	else
		cout<<"\n\n\'"<<find<<"\' Found "<<count<<" Times in The Line...\n\n";
	cout<<"Press Any Key To Continue.....";
	getch();
	return(0);
}
