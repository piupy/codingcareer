#include<iostream>
#include<cstdio>
#include<cstring>
#include<conio.h>
using namespace std;

int main(void)
{
	int n,i,j,k,total=0;
	string country[20];
	cout<<"Enter The Country Names Below. Enter \"X\" To End Reading\n\n";
	for(i=0;;i++)
	{
		cout<<"\n\nCountry "<<(i+1)<<" :    ";
		getline(cin,country[i]);
			if( country[i] == "x" or country[i] == "X" )
			{
				break;
			}
			else
			{
				total++;
			}
	}
	cout<<"\""<<total<<"\" Country Names Successfully Entered...";
		for(j=0;j<total;j++)
		{
			for(k=0;k<total-j;k++)
			{
				if( country[k] > country[k+1] )
				{
					string temp=country[k];
					country[k]=country[k+1];
					country[k+1]=temp;
				}
			}
		}
	cout<<"\n\n";
	for(i=0;i<total;i++)
	{
		cout<<"\n\nCountry "<<(i+1)<<" :    "<<country[i];
	}
	cout<<"\n\nPress Any Key To Continue...";
	getch();
	return(0);
}
