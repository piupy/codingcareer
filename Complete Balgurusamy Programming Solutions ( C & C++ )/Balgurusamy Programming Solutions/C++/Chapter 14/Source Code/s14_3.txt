#include<iostream>
#include<vector>
#include<algorithm>
#include<conio.h>
using namespace std;
typedef vector<int> vec;

void output(vec &v)
{
	int i;
	vec :: iterator itr;
	for(i=0,itr=v.begin();itr!=v.end();itr++,i++)
	{
		cout<<"\n\nElement "<<(i+1)<<" :  "<<v[i];
	}
}

void input(vec &v)
{
	int i;
	vec :: iterator itr;
	for(i=0,itr=v.begin();itr!=v.end();itr++,i++)
	{
		cout<<"\nEnter Element "<<(i+1)<<" :  ";
		cin>>v[i];
	}
}

int main(void)
{
	cout<<"How many Elements You Want To Store :  ";
	int n;
	cin>>n;
	vec v1(n);
	input(v1);
	output(v1);
	cout<<"\n\nWhich Element You Want To Count :  ";
	cin>>n;
	vec :: iterator itr;
	int tell=count(v1.begin(),v1.end(),n);
	cout<<"\n\nTotal Number of Times \""<<n<<"\" Found :  "<<tell;
	cout<<"\n\n\nPress Any Key To Continue...";
	getch();
	return(0);
}
