#include<iostream>
#include<vector>
#include<algorithm>
#include<conio.h>
using namespace std;
typedef vector<int> vec;

void output(vec &v)
{
	int i;
	vec :: iterator itr;
	for(i=0,itr=v.begin();itr!=v.end();itr++,i++)
	{
		cout<<"\n\nElement "<<(i+1)<<" :  "<<v[i];
	}
}

void input(vec &v)
{
	int i;
	vec :: iterator itr;
	for(i=0,itr=v.begin();itr!=v.end();itr++,i++)
	{
		cout<<"\nEnter Element "<<(i+1)<<" :  ";
		cin>>v[i];
	}
}

int main(void)
{
	cout<<"How many Elements You Want To Store :  ";
	int n;
	cin>>n;
	vec v1(n);
	input(v1);
	output(v1);
	cout<<"\n\nWhich Element You Want To Find :  ";
	cin>>n;
	vec :: iterator itr;
	itr=find(v1.begin(),v1.end(),n);
	if(itr!=v1.end())
		cout<<"\n\nFound The Element At Position : "<<(itr-v1.begin()+1);
	else
		cout<<"\n\nNot Found The Element ";
	cout<<"\n\nPress Any Key To Continue...";
	getch();
	return(0);
}
