#include<iostream>
#include<vector>
#include<list>
#include<algorithm>

using namespace std;

class student;

typedef vector<student> vec;

class student
{
	char name[20];
	int roll;
public:
	void input(void);
	void output(void);
	friend void get_name(vec &,int);
	friend void get_roll(vec &,string);
};

void student :: input(void)
{
	cout<<"\n\nEnter The Name of Student :  ";
	cin>>name;
	cout<<"\nEnter The Roll Number of Student :  ";
	cin>>roll;
}

void student :: output(void)
{
	cout<<"\n\nName of Student :  "<<name;
	cout<<"\n\nRoll Number :  "<<roll;
}

void get_name(vec &v,int r)
{
	int flag=0;
	for(int i=0;i<v.size();i++)
	{
		if( v[i].roll == r)
		{
			v[i].output();
			flag=1;
		}
	}
	if(flag==0)
	cout<<"\n\nStudent With Roll Number \""<<r<<"\" Not Found...";
}

void get_roll(vec &v,string r)
{
	int flag=0;
	for(int i=0;i<v.size();i++)
	{
		if( v[i].name == r)
		{
			v[i].output();
			flag=1;
		}
	}
	if(flag==0)
	cout<<"\n\nStudent With Name \""<<r<<"\" Not Found...";
}
