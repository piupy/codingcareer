#include<iostream>
#include<cstdio>
#include<vector>
#include<list>
#include<algorithm>
#include<conio.h>
using namespace std;

class car;

typedef vector<car> vec;

class car
{
	string model_name;
	int units_sold;
	float unit_price;
public:
	void input(void);
	void output(void);
	friend void get_details(vec &,string);
};

void car :: input(void)
{
	cout<<"\n\nEnter The Name of Car :  ";
	cin>>model_name;
	cout<<"\nEnter The Number of Units Sold :  ";
	cin>>units_sold;
	cout<<"\nEnter The Unit Price of Car :  ";
	cin>>unit_price;
}

void car :: output(void)
{
	cout<<"\n\nModel Name :  \""<<model_name<<"\"";
	cout<<"\nUnits Sold :  "<<units_sold<<" units";
	cout<<"\nUnit Price :  Rs."<<unit_price;
	cout<<"\nTotal Sale :  Rs."<<units_sold*unit_price;
}

void get_details(vec &v,string r)
{
	int flag=0;
	for(int i=0;i<v.size();i++)
	{
		if( v[i].model_name == r)
		{
			v[i].output();
			flag=1;
		}
	}
	if(flag==0)
	cout<<"\n\nCar With Model Name \""<<r<<"\" Not Found...";
}

int main(void)
{
	cout<<"How Many Objects You Want To Create :  ";
	int total,i;
	cin>>total;
	cout<<"\n\nCreating "<<total<<" Objects of Class \"Car\" ...";
	car s1;
	vec v1;
	for(i=0;i<total;i++)
	{
		cout<<"\n\n\nEnter The Details of Car Model No. "<<i+1<<" --->\n";
		s1.input();
		v1.push_back(s1);
	}
	vec :: iterator itr;
	for(i=0,itr=v1.begin();i<total;i++,itr++)
	{
		cout<<"\n\n\nCar No. "<<i+1<<" --->\n";
		itr->output();
	}
	string find_name;
	cout<<"\n\nEnter The Name of The Car Model ( Case Sensitive Search ) :  ";
    cin>>find_name;
    get_details(v1,find_name);
    cout<<"\n\n\nPress Any Key To Continue...";
	getch();
	return(0);
}
