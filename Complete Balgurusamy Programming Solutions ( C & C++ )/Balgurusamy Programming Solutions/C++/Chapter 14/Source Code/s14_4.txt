#include<iostream>
#include<vector>
#include<list>
#include<algorithm>
#include<conio.h>
using namespace std;
typedef list<int> vec;

void output(vec &v)
{
	int i;
	vec :: iterator itr;
	for(i=0,itr=v.begin();itr!=v.end();itr++,i++)
	{
		cout<<"\n\nElement "<<(i+1)<<" :  "<<*itr;
	}
}

void input(vec &v)
{
	int i;
	vec :: iterator itr;
	for(i=0,itr=v.begin();itr!=v.end();itr++,i++)
	{
		cout<<"\n\nEnter Element "<<(i+1)<<" :  ";
		cin>>*itr;
	}
}

int main(void)
{
	int n1,n2;
	cout<<"\n\nEnter The Size of Container 1 : ";
	cin>>n1;
	cout<<"\n\nEnter The Size of Container 2 : ";
	cin>>n2;
	vec v1(n1);
	vec v2(n2);
	cout<<"\n\nEnter The Contents of First Container --->";
	input(v1);
	cout<<"\n\nEnter The Contents of Second Container --->";
	input(v2);
	cout<<"\n\nDisplaying The Contents of First Container --->";
	output(v1);
	cout<<"\n\nDisplaying The Contents of Second Container --->";
	output(v2);
	cout<<"\n\nMerging The Two Containers...";
	v1.merge(v2);
	cout<<"\n\nDisplaying The Contents of Merged Container --->";
	output(v1);
	cout<<"\n\n\nPress Any Key To Continue...";
	getch();
	return(0);
}
