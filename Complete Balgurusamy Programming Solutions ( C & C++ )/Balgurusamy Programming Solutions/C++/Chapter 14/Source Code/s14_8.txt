class item;

typedef vector<item> vec;

class item
{
	string item_name;
	int units_sold;
	float unit_price;
public:
	void input(void);
	void output(void);
};

void item :: input(void)
{
	cout<<"\n\nEnter The Name of item :  ";
	cin>>item_name;
	cout<<"\nEnter The Number of Units Sold :  ";
	cin>>units_sold;
	cout<<"\nEnter The Unit Price of Item :  ";
	cin>>unit_price;
}

void item :: output(void)
{
	cout<<"\n\nItem Name :  \""<<item_name<<"\"";
	cout<<"\nUnits Sold :  "<<units_sold<<" units";
	cout<<"\nUnit Price :  Rs."<<unit_price;
	cout<<"\nTotal Sale :  Rs."<<units_sold*unit_price;
}
